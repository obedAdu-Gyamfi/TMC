

\section{What is a Database?}

\begin{frame}{\textcolor{brown}{What is a Database?}}
  \begin{block}{Definition}
    A \textbf{database} is a structured, organised collection of data stored electronically
    in a computer system, designed for efficient storage, retrieval, and management.
  \end{block}

  \vspace{0.5em}
  \textbf{Why choose a database over a spreadsheet?}
  \vspace{0.3em}

  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{itemize}
        \item[\faChartBar] \textcolor{brown}{\textbf{Scalability}} \\
          Handles massive data volumes that would crash or slow a spreadsheet.
        \vspace{0.4em}
        \item[\faCheckCircle] \textcolor{brown}{\textbf{Data Integrity}} \\
          Built-in rules keep data accurate and consistent at all times.
      \end{itemize}

    \column{0.48\textwidth}
      \begin{itemize}
        \item[\faUsers] \textcolor{brown}{\textbf{Concurrency}} \\
          Thousands of users can read and write simultaneously without conflicts.
        \vspace{0.4em}
        \item[\faLock] \textcolor{brown}{\textbf{Security}} \\
          Fine-grained access controls — only authorised users can view or edit data.
      \end{itemize}
  \end{columns}
\end{frame}
