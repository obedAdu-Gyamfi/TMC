\section{Summary}

\begin{frame}{\textcolor{brown}{Summary}}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Key Concepts}
      \begin{itemize}
        \item VCS = change tracking over time
        \item DVCS (Git) vs CVCS (SVN)
        \item Git runs locally; GitHub hosts in the cloud
      \end{itemize}

      \vspace{0.5em}
      \textbf{Essential Commands}
      \begin{itemize}
        \item \texttt{git init} — start a repo
        \item \texttt{git add .} — stage changes
        \item \texttt{git commit -m} — save snapshot
        \item \texttt{git push} — upload to GitHub
      \end{itemize}

    \column{0.48\textwidth}
      \textbf{3 Ways to Connect to GitHub}
      \begin{enumerate}
        \item \textbf{gh CLI} — fastest for new projects
        \item \textbf{Clone} — start from GitHub
        \item \textbf{Remote add} — link an existing local project
      \end{enumerate}

      \vspace{0.8em}
      \begin{block}{Next Steps}
        Explore \texttt{README.md} files, \texttt{.gitignore}, branching (\texttt{git branch}), and pull requests on GitHub.
      \end{block}
  \end{columns}
\end{frame}
